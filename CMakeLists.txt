# See: https://www.npmjs.com/package/cmake-js
cmake_minimum_required(VERSION 2.8)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11")
project(tagio)
include_directories(${CMAKE_JS_INC} include)
file(GLOB SOURCE_FILES "src/*.cc" "src/*.h")
add_library(${PROJECT_NAME} SHARED ${SOURCE_FILES})
set_target_properties(${PROJECT_NAME} PROPERTIES PREFIX "" SUFFIX ".node")
if(WIN32)
    add_library(tag STATIC IMPORTED)
    set_property(TARGET tag PROPERTY IMPORTED_LOCATION taglib/win32/libtag.a)
elseif(WIN64)
    add_library(tag STATIC IMPORTED)
    set_property(TARGET tag PROPERTY IMPORTED_LOCATION taglib/win64/libtag.a)
endif()
target_link_libraries(${PROJECT_NAME} ${CMAKE_JS_LIB} tag)
